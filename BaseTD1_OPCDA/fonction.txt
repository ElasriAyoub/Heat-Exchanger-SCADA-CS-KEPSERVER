private Array ToProperArray(object OPCReturn)
    {// Self-implemented function to converts objects returned by OPC interface into arrays
      IList myilist = OPCReturn as IList;
      if (myilist!=null)
      { // create an array from Ilist
        // Get the type of elements in Ilist
        Type elttype = myilist.GetType().GetElementType();
        // define array with corresponding type and number of elements
        Array tab = Array.CreateInstance(elttype, myilist.Count);
        // Copy the elements fron myilist into tab
        int i = 0;
        foreach(var name in myilist)
        {   tab.SetValue(name, i); i++;    }
        return tab;
      }
      else
      { return null; }

    }
